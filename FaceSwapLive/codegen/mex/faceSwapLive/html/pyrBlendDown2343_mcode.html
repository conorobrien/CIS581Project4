52function [gaussian, laplacian] = pyrBlendDown(image)
53coder.inline('always')
54    
55    kernel = coder.const(conv2([1 4 6 4 1]/16, [1 4 6 4 1]'/16));
56   
57    gaussTemp = imfilter(image, kernel, 'circular', 'conv');
58    gaussian = gaussTemp(1:2:end, 1:2:end,:);
59
60    gaussTemp(:) = 0;
61    gaussTemp(1:2:end, 1:2:end,:) = gaussian;
62    laplacian  = image - imfilter(gaussTemp,4*kernel, 'circular', 'conv');
63end
