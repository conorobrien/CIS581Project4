65function [imageRecon] = pyrBlendUp(image, laplacian)
66coder.inline('always')
67
68    kernel = coder.const(conv2([1 4 6 4 1]/16, [1 4 6 4 1]'/16));
69
70    imageRecon = zeros(size(laplacian,1), size(laplacian,2), 3);
71    imageRecon(1:2:end, 1:2:end,:) = image;
72    imageRecon = imfilter(imageRecon, 4*kernel, 'circular', 'conv') + laplacian;
73end
