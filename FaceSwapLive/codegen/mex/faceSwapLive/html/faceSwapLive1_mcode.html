1function swappedImage = faceSwapLive(targetImage, swapImage, swapMask)
 2    scale = 1/6; % shrink scale for swapImage
 3
 4    persistent faceDetector rEyeDetector lEyeDetector mouthDetector noseDetector;
 5
 6    %only declare the detectors the first time function is called, easier to deal with coder that way
 7    if isempty(faceDetector) 
 8        faceDetector = vision.CascadeObjectDetector('FrontalFaceCART');
 9        lEyeDetector = vision.CascadeObjectDetector('lefteye');
10        rEyeDetector = vision.CascadeObjectDetector('righteye');
11        mouthDetector = vision.CascadeObjectDetector('mouth');
12        noseDetector = vision.CascadeObjectDetector('nose');
13    end
14    
15    % Detect faces in image
16    faceBoxes = step(faceDetector, targetImage);
17    
18    % Initialize swappedImage, if no faces are found it just returns targetImage
19    swappedImage = targetImage;
20    
21    for j = 1:size(faceBoxes, 1)
22        
23        xRange = (1:faceBoxes(j,4)) + faceBoxes(j,2);
24        yRange = (1:faceBoxes(j,3)) + faceBoxes(j,1);
25        faceImage = targetImage(xRange, yRange,:);
26
27        coder.varsize('faceImageGray', [200 200]); %coder needs to know max size of face
28        faceImageGray = rgb2gray(faceImage); %coder step needs grayscale image
29
30        % detect features in face
31        rEyeBoxes = step(rEyeDetector, faceImageGray);
32        lEyeBoxes = step(lEyeDetector, faceImageGray);
33        mouthBoxes = step(mouthDetector, faceImageGray);
34        noseBoxes = step(noseDetector, faceImageGray);
35        
36        % If anything is missing, skip this facebox
37        if isempty(rEyeBoxes) || isempty(lEyeBoxes) || isempty(mouthBoxes) || isempty(noseBoxes)
38            continue
39        end
40        
41        % Determine the best features using the pictoral structure, k-fan
42        bestPoints = featSelectLive(noseBoxes, mouthBoxes, lEyeBoxes, rEyeBoxes);
43        basePoints = double([324 390 200 485 460 485 211 235 421 226]*scale);
44        
45        % If featSelectLive rejects points, skip this facebox
46        if isempty(bestPoints)
47            continue
48        end
49                
50        % Reshape control points for warping
51        swapPts = reshapePoints(basePoints);
52        targetPts = reshapePoints(bestPoints);
53        
54        % Fit transformation and warp image and mask
55        tform = fitgeotrans(swapPts, targetPts, 'projective');
56        warpedImage = imwarp(swapImage, tform,'nearest', 'OutputView', imref2d(size(faceImage)));
57        warpedMask = imwarp(swapMask, tform,'nearest', 'OutputView', imref2d(size(faceImage)));
58        
59        imageMask = false(size(targetImage(:,:,1)));
60        imageMask(xRange, yRange) = warpedMask;
61        
62        morphedBlend = targetImage;
63        morphedBlend(xRange, yRange,:) = warpedImage;
64
65        swappedImage = pyramidBlendLive(morphedBlend, targetImage, imageMask);
66
67        if size(faceBoxes, 1) > 1
68            targetImage = swappedImage;
69        end
70    end
71end
