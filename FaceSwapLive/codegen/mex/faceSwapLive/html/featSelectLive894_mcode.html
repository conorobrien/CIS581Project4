1function [bestPoints] = featSelectLive(noseBoxes, mouthBoxes, lEyeBoxes, rEyeBoxes)
 2% possible improvements:
 3% - use left/right of mouth rather then mouth center
 4% - needs speedup, the pointsDist function gets really slow as # features increases
 5scale = 1/3/2;
 6% Pull out center of boxes and count each feature
 7noseCenter = boxCenter(noseBoxes);
 8numNose = size(noseCenter,1);
 9mouthCenter = boxCenter(mouthBoxes);
10numMouth = size(mouthCenter,1);
11mouthLeft = [mouthBoxes(:,1) mouthBoxes(:,2)+mouthBoxes(:,4)/2];
12mouthRight = [mouthBoxes(:,1)+mouthBoxes(:,3) mouthBoxes(:,2)+mouthBoxes(:,4)/2];
13% numMouth = size(mouthLeft,1);
14lEyeCenter = boxCenter(lEyeBoxes);
15numLEye = size(lEyeCenter,1);
16rEyeCenter = boxCenter(rEyeBoxes);
17numREye = size(rEyeCenter,1);
18
19% Create matrix with permutation of every feature as a row
20lTotal = numNose*numMouth*numLEye*numREye;
21idxNose = repmat(1:numNose, lTotal/numNose,1);
22idxMouth = repmat(1:numMouth,lTotal/numNose/numMouth, numNose);
23idxLEye = repmat(1:numLEye,lTotal/numNose/numMouth/numLEye,numNose*numMouth);
24idxREye = repmat(1:numREye,1,lTotal/numREye);
25
26% Feature matrix, points = [x1 y1 x2 y2 x3 y3 x4 y4]
27points = [noseCenter(idxNose(:), :) mouthLeft(idxMouth(:), :) mouthRight(idxMouth(:), :) ...
28    lEyeCenter(idxLEye(:), :) rEyeCenter(idxREye(:), :)];
29
30% Points of my test face
31basePoints = double([324 390 200 485 460 485 211 235 421 226]*scale);
32
33% k-Fan distances for base face
34baseDist = pointsDist(basePoints);
35
36% This is pretty slow, try to vectorize it later
37dists = zeros(lTotal, 16);
38for i = 1:lTotal
39    dists(i,:) = pointsDist(points(i,:));
40end
41
42% Find the closest in k-fan space (using l1 norm now, easier to 1line)
43[minDist, minIdx] = min(sum(abs(bsxfun(@minus, dists, baseDist)),2));
44
45if minDist > 2.5
46    bestPoints = [];
47else
48    % Return points, the index in boxes, and the homography (don't use homography anymore though)
49    bestPoints = round(points(minIdx,:));
50
51    % Calculate the left and right corners of the mouth
52    bestMouthC = mouthCenter(idxMouth(minIdx),:);
53    bestMouth = mouthBoxes(idxMouth(minIdx),:);
54    bestREye = bestPoints(7:8);
55    bestLEye = bestPoints(9:10);
56
57    mouthLeft = bestMouthC - (bestREye - bestLEye)/(bestREye(1) - bestLEye(1))*(bestMouth(3)/2);
58    mouthRight = bestMouthC + (bestREye - bestLEye)/(bestREye(1) - bestLEye(1))*(bestMouth(3)/2);
59
60    if any(isfinite(mouthLeft) & isfinite(mouthRight))
61        bestPoints(3:4) = mouthLeft;
62        bestPoints(5:6) = mouthRight;
63    end
64end
65
66end
