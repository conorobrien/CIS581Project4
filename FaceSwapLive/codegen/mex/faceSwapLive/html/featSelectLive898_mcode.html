1function [bestPoints] = featSelectLive(noseBoxes, mouthBoxes, lEyeBoxes, rEyeBoxes)
 2coder.inline('always')
 3
 4scale = 1/6; %scale of baseImage, so I don't have to keep resizing basePoints
 5
 6% Pull out center of boxes (and LR of mouth) and count each feature
 7noseCenter = boxCenter(noseBoxes);
 8numNose = size(noseCenter,1);
 9mouthLeft = [mouthBoxes(:,1) mouthBoxes(:,2)+mouthBoxes(:,4)/2];
10mouthRight = [mouthBoxes(:,1)+mouthBoxes(:,3) mouthBoxes(:,2)+mouthBoxes(:,4)/2];
11numMouth = size(mouthLeft,1);
12lEyeCenter = boxCenter(lEyeBoxes);
13numLEye = size(lEyeCenter,1);
14rEyeCenter = boxCenter(rEyeBoxes);
15numREye = size(rEyeCenter,1);
16
17% Create matrix with permutation of every feature as a row
18lTotal = numNose*numMouth*numLEye*numREye;
19idxNose = repmat(1:numNose, lTotal/numNose,1);
20idxMouth = repmat(1:numMouth,lTotal/numNose/numMouth, numNose);
21idxLEye = repmat(1:numLEye,lTotal/numNose/numMouth/numLEye,numNose*numMouth);
22idxREye = repmat(1:numREye,1,lTotal/numREye);
23
24% Feature matrix, points = [x1 y1 x2 y2 x3 y3 x4 y4]
25points = [noseCenter(idxNose(:), :) mouthLeft(idxMouth(:), :) mouthRight(idxMouth(:), :) ...
26    lEyeCenter(idxLEye(:), :) rEyeCenter(idxREye(:), :)];
27
28% Points of my test face
29basePoints = double([324 390 200 485 460 485 211 235 421 226]*scale);
30
31% k-Fan distances for base face
32baseDist = pointsDist(basePoints);
33
34% This is pretty slow, try to vectorize it later
35dists = zeros(lTotal, 16);
36for i = 1:lTotal
37    dists(i,:) = pointsDist(points(i,:));
38end
39
40% Find the closest in k-fan space
41[minDist, minIdx] = min(sum(abs(bsxfun(@minus, dists, baseDist)),2));
42
43% If minimum is too far away, reject all faces, otherwise retrun the best
44if minDist > 4
45    bestPoints = [];
46else
47    % Return points, the index in boxes, and the homography (don't use homography anymore though)
48    bestPoints = round(points(minIdx,:));
49
50    % Calculate the left and right corners of the mouth
51    bestMouth = mouthBoxes(idxMouth(minIdx),:);
52    bestMouthC = boxCenter(bestMouth);
53    bestREye = bestPoints(7:8);
54    bestLEye = bestPoints(9:10);
55
56    mouthLeft = bestMouthC - (bestREye - bestLEye)/(bestREye(1) - bestLEye(1))*(bestMouth(3)/2);
57    mouthRight = bestMouthC + (bestREye - bestLEye)/(bestREye(1) - bestLEye(1))*(bestMouth(3)/2);
58
59    if any(isfinite(mouthLeft) & isfinite(mouthRight)) % Reject NaN or Inf points, it happens sometimes
60        bestPoints(3:4) = mouthLeft;
61        bestPoints(5:6) = mouthRight;
62    end
63end
64
65end
